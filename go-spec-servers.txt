package server_test

import (
	"context"
	"encoding/json"
	"fmt"
	"io"
	"log"
	"os"

	"github.com/tmc/mcp/protocol"
	"github.com/tmc/mcp/server"
	"github.com/tmc/mcp/transport/stdio"
)

type EchoTool struct{}

func (t *EchoTool) Name() string {
	return "echo"
}

func (t *EchoTool) Description() string {
	return "Echoes the input message."
}

func (t *EchoTool) InputSchema() json.RawMessage {
	return json.RawMessage(`{"type": "object", "properties": {"message": {"type": "string"}}, "required": ["message"]}`)
}

func (t *EchoTool) OutputSchema() json.RawMessage {
	return json.RawMessage(`{"type": "object", "properties": {"message": {"type": "string"}}}`)
}

func (t *EchoTool) Handle(ctx context.Context, req *protocol.Request) (*protocol.Response, error) {
	var params struct {
		Message string `json:"message"`
	}
	if err := json.Unmarshal(req.Params, &params); err != nil {
		return protocol.NewErrorResponse(req.ID, protocol.InvalidParams, "Invalid parameters"), nil
	}

	result := struct {
		Message string `json:"message"`
	}{
		Message: params.Message,
	}
	resultData, _ := json.Marshal(result)

	return &protocol.Response{
		ID:      req.ID,
		Result:  resultData,
		JSONRPC: protocol.JSONRPCVersion,
	}, nil
}

func ExampleServer() {
	// Create a new server
	s := server.New("example-server", "0.1.0")

	// Register the echo tool
	echoTool := &EchoTool{}
	if err := s.RegisterTool(echoTool); err != nil {
		log.Fatalf("Failed to register tool: %v", err)
	}

	// Create a transport (using stdio for this example)
	transport := stdio.NewTransport(os.Stdin, os.Stdout)

	// Create a context that can be used to stop the server
	ctx, cancel := context.WithCancel(context.Background())
	defer cancel()

	// Start serving requests in a separate goroutine
	go func() {
		if err := s.Serve(ctx, transport); err != nil {
			if err != io.EOF {
				log.Fatalf("Server error: %v", err)
			}
		}
	}()

	fmt.Println("Server started. Press Ctrl+C to stop.")

	// Keep the main goroutine alive until interrupted
	select {}
}

package http_test

import (
	"context"
	"fmt"
	"log"
	"net"
	"net/http"
	"time"

	"github.com/tmc/mcp/client"
	"github.com/tmc/mcp/protocol"
	"github.com/tmc/mcp/server"
	"github.com/tmc/mcp/transport/http"
)

func ExampleTransport() {
	// Create a listener for the HTTP server
	listener, err := net.Listen("tcp", "127.0.0.1:0") // Bind to an available port
	if err != nil {
		log.Fatalf("Failed to create listener: %v", err)
	}
	defer listener.Close()

	// Get the address of the listener
	addr := listener.Addr().String()
	fmt.Printf("Listening on http://%s\n", addr)

	// Create a new HTTP transport for the server
	httpTransport := http.NewTransport(listener)

	// Create a new MCP server
	s := server.New("example-server", "0.1.0")

	// Start serving in a separate goroutine
	go func() {
		if err := s.Serve(context.Background(), httpTransport); err != nil && err != http.ErrServerClosed {
			log.Fatalf("Server error: %v", err)
		}
	}()

	// Create a new HTTP transport for the client
	clientTransport := http.NewClientTransport(fmt.Sprintf("http://%s/", addr))
	defer clientTransport.Close()

	// Create a new MCP client
	c := client.New(clientTransport)
	defer c.Close()

	// Initialize the client with a timeout
	ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
	defer cancel()

	initResult, err := c.Initialize(ctx, &protocol.InitializeParams{
		ClientInfo: protocol.ClientInfo{
			Name:    "example-client",
			Version: "0.1.0",
		},
		Capabilities: protocol.ClientCapabilities{},
	})
	if err != nil {
		log.Fatalf("Failed to initialize client: %v", err)
	}

	fmt.Printf("Server Info: %s (version %s)\n", initResult.ServerInfo.Name, initResult.ServerInfo.Version)

	// Output:
	// Listening on http://127.0.0.1:53178
	// Server Info: example-server (version 0.1.0)
}

package main

import (
	"context"
	"encoding/json"
	"flag"
	"fmt"
	"log"
	"os"
	"strconv"

	"github.com/tmc/mcp"
	"github.com/tmc/mcp/protocol"
	"github.com/tmc/mcp/server"
	"github.com/tmc/mcp/transport/stdio"
)

func main() {
	var (
		name    = flag.String("name", "calculator-server", "name of the server")
		version = flag.String("version", "0.1.0", "version of the server")
	)
	flag.Usage = func() {
		fmt.Fprintf(os.Stderr, "Usage: %s [flags]\n", os.Args[0])
		flag.PrintDefaults()
	}
	flag.Parse()

	// Create a new MCP service
	svc := mcp.NewService(*name, *version)

	// Register the calculator tool
	if err := svc.RegisterTool(NewCalculatorTool()); err != nil {
		log.Fatalf("Failed to register calculator tool: %v", err)
	}

	// Create a new server
	server := server.New(svc)

	// Serve on stdin/stdout
	transport := stdio.NewTransport(context.Background())
	if err := server.Serve(context.Background(), transport); err != nil {
		log.Fatalf("Server error: %v", err)
	}
}

// CalculatorTool implements a simple calculator.
type CalculatorTool struct{}

// NewCalculatorTool creates a new CalculatorTool instance.
func NewCalculatorTool() *CalculatorTool {
	return &CalculatorTool{}
}

// Name returns the name of the tool.
func (t *CalculatorTool) Name() string {
	return "calculate"
}

// Description returns the description of the tool.
func (t *CalculatorTool) Description() string {
	return "Performs basic arithmetic calculations."
}

// InputSchema returns the JSON schema for the tool's input parameters.
func (t *CalculatorTool) InputSchema() json.RawMessage {
	return json.RawMessage(`{
		"type": "object",
		"properties": {
			"expression": {"type": "string", "description": "Arithmetic expression to evaluate"}
		},
		"required": ["expression"]
	}`)
}

// OutputSchema returns the JSON schema for the tool's output.
func (t *CalculatorTool) OutputSchema() json.RawMessage {
	return json.RawMessage(`{
		"type": "object",
		"properties": {
			"result": {"type": "number", "description": "Result of the calculation"}
		}
	}`)
}

// Handler executes the calculator tool.
func (t *CalculatorTool) Handler(ctx context.Context, req *protocol.Request) (*protocol.Response, error) {
	var params struct {
		Expression string `json:"expression"`
	}
	if err := json.Unmarshal(req.Params, &params); err != nil {
		return protocol.NewErrorResponse(req.ID, protocol.InvalidParams, "Invalid parameters"), nil
	}

	result, err := evaluateExpression(params.Expression)
	if err != nil {
		return protocol.NewErrorResponse(req.ID, protocol.InternalError, fmt.Sprintf("Failed to evaluate expression: %v", err)), nil
	}

	resultData, _ := json.Marshal(struct {
		Result float64 `json:"result"`
	}{
		Result: result,
	})

	return &protocol.Response{
		ID:      req.ID,
		Result:  resultData,
		JSONRPC: protocol.JSONRPCVersion,
	}, nil
}

// evaluateExpression evaluates a simple arithmetic expression.
func evaluateExpression(expression string) (float64, error) {
	// This is a placeholder for a proper expression evaluator.
	// For simplicity, we just return the parsed number.
	// In a real implementation, you would parse and evaluate the expression.
	return strconv.ParseFloat(expression, 64)
}
