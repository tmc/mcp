// Package mcp provides a Go implementation of the Model Context Protocol (MCP).
//
// MCP is an open protocol that enables seamless integration between LLM applications and external data sources and tools.
// This package defines the core data structures, interfaces, and functions for building MCP clients and servers.
//
// The implementation is organized into several subpackages:
//
// - `client`: Provides functionality for creating and managing MCP clients.

import (
	"context"

	"fmt"

	"log"
	"os"

	"github.com/tmc/mcp/protocol"
	"github.com/tmc/mcp/server"
	"github.com/tmc/mcp/transport/stdio"
)

type EchoTool struct{}

func (t *EchoTool) Name() string {
	return "echo"
}

func (t *EchoTool) Description() string {
	return "Echoes the input message."
}

func (t *EchoTool) InputSchema() json.RawMessage {
	return json.RawMessage(`{"type": "object", "properties": {"message": {"type": "string"}}, "required": ["message"]}`)
}

func (t *EchoTool) OutputSchema() json.RawMessage {
	return json.RawMessage(`{"type": "object", "properties": {"message": {"type": "string"}}}`)
}

func (t *EchoTool) Handle(ctx context.Context, req *protocol.Request) (*protocol.Response, error) {
	var params struct {
		Message string `json:"message"`
	}
	if err := json.Unmarshal(req.Params, &params); err != nil {
		return protocol.NewErrorResponse(req.ID, protocol.InvalidParams, "Invalid parameters"), nil
	}

	result := struct {
		Message string `json:"message"`
	}{
		Message: params.Message,
	}
	resultData, _ := json.Marshal(result)

	return &protocol.Response{
		ID:      req.ID,
		Result:  resultData,
		JSONRPC: protocol.JSONRPCVersion,
	}, nil
}

func ExampleServer() {
	// Create a new server
	s := server.New("example-server", "0.1.0")

	// Register the echo tool
	echoTool := &EchoTool{}
	if err := s.RegisterTool(echoTool); err != nil {
		log.Fatalf("Failed to register tool: %v", err)
	}

	// Create a transport (using stdio for this example)
	transport := stdio.NewTransport(os.Stdin, os.Stdout)

	// Create a context that can be used to stop the server
	ctx, cancel := context.WithCancel(context.Background())
	defer cancel()

	// Start serving requests in a separate goroutine
	go func() {
		if err := s.Serve(ctx, transport); err != nil {
			if err != io.EOF {
				log.Fatalf("Server error: %v", err)
			}
		}
	}()

	fmt.Println("Server started. Press Ctrl+C to stop.")

	// Keep the main goroutine alive until interrupted
	select {}
}

import (
	"context"
	"fmt"
	"log"
	"net"
	"net/http"
	"time"

	"github.com/tmc/mcp/client"
	"github.com/tmc/mcp/protocol"
	"github.com/tmc/mcp/server"
	"github.com/tmc/mcp/transport/http"
)

func ExampleTransport() {
	// Create a listener for the HTTP server
	listener, err := net.Listen("tcp", "127.0.0.1:0") // Bind to an available port
	if err != nil {
		log.Fatalf("Failed to create listener: %v", err)
	}
	defer listener.Close()

	// Get the address of the listener
	addr := listener.Addr().String()
	fmt.Printf("Listening on http://%s\n", addr)

	// Create a new HTTP transport for the server
	httpTransport := http.NewTransport(listener)

	// Create a new MCP server
	s := server.New("example-server", "0.1.0")

	// Start serving in a separate goroutine
	go func() {
		if err := s.Serve(context.Background(), httpTransport); err != nil && err != http.ErrServerClosed {
			log.Fatalf("Server error: %v", err)
		}
	}()

	// Create a new HTTP transport for the client
	clientTransport := http.NewClientTransport(fmt.Sprintf("http://%s/", addr))
	defer clientTransport.Close()

	// Create a new MCP client
	c := client.New(clientTransport)
	defer c.Close()

	// Initialize the client with a timeout
	ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
	defer cancel()

	initResult, err := c.Initialize(ctx, &protocol.InitializeParams{
		ClientInfo: protocol.ClientInfo{
			Name:    "example-client",
			Version: "0.1.0",
		},
		Capabilities: protocol.ClientCapabilities{},
	})
	if err != nil {
		log.Fatalf("Failed to initialize client: %v", err)
	}

	fmt.Printf("Server Info: %s (version %s)\n", initResult.ServerInfo.Name, initResult.ServerInfo.Version)

	// Output:
	// Listening on http://127.0.0.1:53178
	// Server Info: example-server (version 0.1.0)
}

// ClientTransport is a client-side transport using HTTP and SSE.
type ClientTransport struct {
	serverURL string
	client    *http.Client
	sseClient *sse.Client
	events    chan *sse.Event
	cancel    context.CancelFunc
}

// NewClientTransport creates a new client-side HTTP transport.
func NewClientTransport(serverURL string) *ClientTransport {
	ctx, cancel := context.WithCancel(context.Background())
	c := &ClientTransport{
		serverURL: serverURL,
		client:    &http.Client{},
		sseClient: sse.NewClient(serverURL),
		events:    make(chan *sse.Event),
		cancel:    cancel,
	}

	// Subscribe to the SSE stream
	err := c.sseClient.SubscribeWithContext(ctx, "messages", func(msg *sse.Event) {
		// Filter out non-data events (e.g., heartbeat)
		if len(msg.Data) > 0 {
			c.events <- msg
		}
	})
	if err != nil {
		fmt.Printf("Error subscribing to SSE stream: %v\n", err)
	}

	return c
}

// Read reads a message from the server.
func (t *ClientTransport) Read(p []byte) (n int, err error) {
	event := <-t.events
	n = copy(p, event.Data)
	return
}

// Write writes a message to the server.
func (t *ClientTransport) Write(p []byte) (n int, err error) {
	req, err := http.NewRequest("POST", t.serverURL, bytes.NewReader(p))
	if err != nil {
		return 0, err
	}
	req.Header.Set("Content-Type", "application/json")

	resp, err := t.client.Do(req)
	if err != nil {
		return 0, err
	}
	defer resp.Body.Close()

	if resp.StatusCode != http.StatusOK {
		return 0, fmt.Errorf("HTTP error: %s", resp.Status)
	}

	// Read the response body to avoid resource leaks
	_, err = io.Copy(io.Discard, resp.Body)
	if err != nil {
		return 0, fmt.Errorf("failed to read response body: %w", err)
	}

	return len(p), nil
}

// Close closes the connection.
func (t *ClientTransport) Close() error {
	t.cancel()
	return nil
}

# Test Tool Specification

The `test` tool is a standard tool in the Model Context Protocol (MCP) for verifying the behavior of MCP implementations. It allows clients to execute predefined test scripts or interact with a server in a controlled manner to ensure conformance to the MCP specification.

## Input Parameters

| Parameter   | Type     | Optional | Default | Description                                                                                                                                                                                                 |
| :---------- | :------- | :------- | :------ | :---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------- |
| `script`    | `string` | Yes      |         | The path to a test script file. The format of the test script can be `txtar`, `YAML`, or a custom format defined by the server. If not specified, the tool may enter an interactive mode or use a default script. |
| `server`    | `string` | Yes      |         | The address of the MCP server to test. This can be a URL for HTTP transport or a path for stdio transport. If not specified, the tool may start a local server instance for testing.                     |
| `timeout`   | `number` | Yes      | `60`    | The timeout in seconds for each test case or interaction.                                                                                                                                                 |
| `verbose`   | `boolean`| Yes      | `false` | Whether to enable verbose output, including detailed information about each test step.                                                                                                                    |
| `failfast`  | `boolean`| Yes      | `false` | Whether to stop the test execution on the first failure.                                                                                                                                                  |

## Output

| Field    | Type     | Optional | Description                                                                                                |
| :------- | :------- | :------- | :--------------------------------------------------------------------------------------------------------- |
| `result` | `string` | No       | The overall result of the test execution: "success" or "failure".                                           |
| `details`| `array`  | Yes      | A list of detailed test results, including the status, duration, and any error messages for each test case. |
| `output` | `string` | Yes      | The combined output (stdout and stderr) of the test execution.                                             |

## Modes of Operation

The `test` tool supports the following modes of operation:

### Script Mode

In script mode, the tool executes a test script specified by the `script` parameter. The script can contain a sequence of commands, assertions, and control flow statements to define the test scenario. The tool parses the script, executes the commands, and verifies the assertions.

### Interactive Mode

In interactive mode, the tool starts an interactive session with the server, allowing the user to send requests and receive responses manually. This mode can be useful for debugging or exploratory testing. If no script is provided, the tool may enter interactive mode by default.

## Error Codes

The `test` tool may return the following error codes:

| Code | Description                                                     |
| :--- | :-------------------------------------------------------------- |
| `1`  | Indicates that one or more test cases failed.                  |
| `2`  | Indicates an error in the test script syntax or format.         |
| `3`  | Indicates a connection error or timeout during communication. |
| `4`  | Indicates an invalid or unsupported parameter value.            |
